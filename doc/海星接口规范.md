### 1 海星接口规范*
#### 代码
##### 公用
>* 400  成功
>* 401  token过期，重新登入 
>* 402  该课程已发布
>* 403  手机号已注册
>* 405  密码错误
>* 406  数据库错误

```php
Param
    token      //token
    
    c_img    //课程封面jpg or png
    c_name   //课程名称
    c_major  //课程专业
    c_time   //课程时间
    c_price  //课程价格
    c_place  //课程地点
    c_detail //课程详情
    tags
    [
        多个标签
    ]

Function cou_release()
{
    //check token
    if(token过期)
    {
        return 401
    }
    
    update token
    
    通过token返回u_id
    
    /*
    *insert courses_1、courses_2
    *先通过u_id搜索到该用户全部课程
    *再对比，会比较快
    *
    */
    if(c_major,c_name，c_detail与数据库中的不同)
    {
        insert into courses_1 courses_2,tags where u_id = u_id
    }else
    {
        return 402
    }
    
    
    if(数据库错误)
    {
        return 406
    }
    
    return 400 
}
```

#### 5
##### 场景
>* 卖家界面-点击进入该课程

##### 接口

```php
Param
    token    //token
    c_id     //课程ID
Funtion cou_entercou()
{
     //check token
    if(token过期)
    {
        return 401
    }
    
    update token
    /**
    *通过c_id搜索order,累加order_state=4的order_money得到总收益c_revenue
    *通过u_id搜索courses_2,累加c_purchase得到总销量c_totalsales
    *通过u_id搜索courses_2,平均所有课程的c_star得到总满意度c_totalstar
    *通过u_id搜索courses_2,得到买家发布的总课程数c_sum
    *通过u_id搜索users_2,u_birth,用当前年份减去得到年龄u_age（暂时，后期通过身份证）
    */
    select c_id,c_imgpath,c_name,c_major,c_time,c_price,c_intro,c_detail,c_start,tag_text
    from courses_1,courses_2
    select u_sex,u_birth,u_major from users_2
    $data['c_info'] = array('c_id','c_imgpath','c_name','c_major','c_time','c_price',
                            'c_intro','c_detail','c_start','tag_text','c_revenue');
    $data['u_info'] = array('u_sex','u_birth','u_major',
                            'c_totalsales','c_totalstar','c_sum')
    
    return  $data;
}
```
